/* 力士ID=5 のみのテーブルのビューを作成 */
/* CREATE VIEW CONTRIBUTIONS_RIKISHI_ID_5 AS SELECT * from CONTRIBUTIONS WHERE rikishi_id = 5; */

SELECT * FROM CONTRIBUTIONS_RIKISHI_ID_5;

/* 力士ID=5 の投稿にSTARの有無を付けたものを表示。STARの欄がNULLだとSTAR無しとなっている */
SELECT C_5.*, S.ID AS STAR
FROM CONTRIBUTIONS_RIKISHI_ID_5 AS C_5 LEFT OUTER JOIN STARS AS S
ON C_5.id = S.CONTRIBUTION_id;
