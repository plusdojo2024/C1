/*USERSテーブルを作成する SQL 文*/

CREATE TABLE USERS (
ID		VARCHAR (50) ,
USER_ID		VARCHAR (50) NOT NULL,
PASSWORD	VARCHAR (20) NOT NULL,
USER_NAME	VARCHAR (30) NOT NULL,
ICON		VARCHAR (100),
MESSAGE		VARCHAR (200),
ASK		VARCHAR (20) NOT NULL,
QUESTION	VARCHAR (20) NOT NULL,
CREATED_AT	TIMESTAMP,
UPDATED_AT	TIMESTAMP,
PRIMARY KEY (ID));
