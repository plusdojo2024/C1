/*followのSQL*/
SELECT U.user_id, U.user_name, U.icon, F.FOLLOW_USER_ID AS FOLLOWS_USER_NAME
FROM (SELECT * from USERS WHERE ID = 3) AS U LEFT OUTER JOIN FOLLOWS AS F
ON U.user_id = F.user_id;
